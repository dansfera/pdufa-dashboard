        function calculateDaysRemaining(dateStr) {

            const pdufaDate = new Date(dateStr);

            const today = new Date();

            const diffTime = pdufaDate - today;

            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));

            return diffDays;

        }


        function getDaysClass(days) {

            if (days <= 7) return 'days-urgent';

            if (days <= 30) return 'days-soon';

            return 'days-future';

        }


        function formatDate(dateStr) {

            const date = new Date(dateStr);

            return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });

        }


        function getAreaClass(area) {

            const map = {

                'oncology': 'ta-oncology',

                'rare': 'ta-rare',

                'immunology': 'ta-immunology',

                'endocrine': 'ta-endocrine',

                'gene': 'ta-gene',

                'other': 'ta-other'

            };

            return map[area] || 'ta-other';

        }


        function getAreaLabel(area) {

                'gene': 'Gene Therapy',

            };

            return map[area] || area;

        }


        function renderTable(filter = 'all') {

            const tbody = document.getElementById('pdufa-body');

            tbody.innerHTML = '';

        function updateStats() {

            const today = new Date();

            const thisMonth = today.getMonth();

            const thisYear = today.getFullYear();


            let thisMonthCount = 0;

            let next30Count = 0;

            let highPriorityCount = 0;


            pdufaData.forEach(item => {

                const date = new Date(item.date);

                const days = calculateDaysRemaining(item.date);


                if (date.getMonth() === thisMonth && date.getFullYear() === thisYear) {

                    thisMonthCount++;

                }

                if (days <= 30 && days >= 0) {

                    next30Count++;

                }

                if (item.priority === 'high') {

                    highPriorityCount++;

                }

            });


            document.getElementById('total-count').textContent = pdufaData.length;

            document.getElementById('this-month').textContent = thisMonthCount;

            document.getElementById('next-30').textContent = next30Count;

            document.getElementById('high-priority').textContent = highPriorityCount;

        }
